for

Example

// JavaScript program to illustrate for loop
let x;
// for loop begins when x = 2
// and runs till x <= 4
for (x = 2; x <= 4; x++) {
console.log("Value of x: " + x);
}

while

Example

// JavaScript code to use while loop
let val = 1;
while (val < 6) {
console.log(val);
val += 1;
}

do-while

Example

let test = 1;
do {
console.log(test);
test++;
} while(test <= 5)

for-in

Example

// JavaScript program to illustrate for loop
let myObj = { x: 1, y: 2, z: 3 };
for (let key in myObj) {
console.log(key, myObj[key]);
}

for-of

Example

let arr = [1, 2, 3, 4, 5];
for (let value of arr) {
console.log(value);
}

Continue Statement

Example

for (let i = 0; i < 11; i++) {
if (i % 2 == 0)
continue;
console.log(i);
}
